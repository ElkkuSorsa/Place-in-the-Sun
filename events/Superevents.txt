add_namespace = superevents
news_event = {
	id = superevents.1
	title = "The Great War"
	desc = "Skibadi"

	fire_only_once = yes
	major = yes
	hidden = yes

	trigger = {
		GER = {
			has_war_with = FRA
		}
	}

	immediate = {
		every_country = {
			if = {
				limit = { NOT = { has_country_flag = SE_superevents_enabled } }
				set_country_flag = SE_superevents_activate
				set_country_flag = SE_ww1
				hidden_effect = {
					play_song = "SE_Kerenski"
				}
			}
		}
	}
}
